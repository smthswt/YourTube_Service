/* global chrome */

async function injectNewVideos() {
    var existingElement = document.querySelector('.style-scope ytd-rich-grid-renderer');

    if (existingElement) {
        // 기존 요소의 스타일 수정
        existingElement.style.width = '95%'; // 가로 크기
        existingElement.style.height = '100%'; // 세로 크기
        existingElement.style.backgroundColor = 'black'; // 배경색 검정으로 설정
        existingElement.style.color = 'white'; // 텍스트 색상
        existingElement.style.fontSize = '30px'; // 글씨 크기
        existingElement.style.display = 'block'; // 블록 레이아웃 사용
        existingElement.style.position = 'relative'; // 고정 위치 설정 해제
        existingElement.style.border = '10px solid white'; // 흰색 테두리 설정
        existingElement.style.zIndex = '1000'; // 다른 요소 위에 표시
        existingElement.style.overflowY = 'auto'; // 세로 스크롤 가능하게 설정
        existingElement.style.padding = '20px'; // 내부 패딩 추가

        // 텍스트 추가
        existingElement.innerText = 'hello world\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\naaa\nbbb\nㅊㅊㅊ';

    } else {
        console.log('선택한 요소를 찾을 수 없습니다.');
        chrome.runtime.sendMessage({action: "forceReload"});
    }
}

// 함수 호출
injectNewVideos();
